\section*{Objectives}
\begin{itemize}[leftmargin=*,nosep]
  \item Use an IDE (e.g.\ Dev-C++, Code::Blocks, VS Code) to create, compile, and run a C program.
  \item Read the structure of a minimal C program (\texttt{\#include}, \texttt{main}, \texttt{return}).
  \item Perform basic arithmetic in expressions and assign results to variables.
  \item Use \texttt{printf} to display output and \texttt{scanf} to read input.
\end{itemize}

\begin{summary}[Quick Reference]
\textbf{Minimal program.}
\begin{lstlisting}
#include <stdio.h>

int main(void) {
    printf("Hello, world!\n");
    return 0;
}
\end{lstlisting}

\textbf{Common format specifiers.}\\
\begin{tabular}{@{}llll@{}}
\toprule
\texttt{\%d} & \texttt{int} & \texttt{\%f} & \texttt{float} \\
\texttt{\%lf} & \texttt{double} & \texttt{\%c} & \texttt{char} \\
\texttt{\%s} & string (\texttt{char[]}) & \texttt{\%\%} & literal \% \\
\bottomrule
\end{tabular}

\textbf{Input.} Use \texttt{scanf("\%d", \&x);} -- the \texttt{\&} (address-of) is required for numeric inputs.
For a single character after a previous \texttt{scanf}, use \texttt{" \%c"} (with leading space) to skip the leftover newline.

\textbf{Workflow.} Edit $\rightarrow$ Save as \texttt{name.c} $\rightarrow$ Compile $\rightarrow$ Run.
A compile error means your source did not produce an executable; read the first error message first.
\end{summary}

\subsection*{Quick Experiments (Try It)}

\begin{tryit}{What happens with a wrong format specifier?}
Type and run this program exactly as shown:
\begin{lstlisting}
#include <stdio.h>
int main(void) {
    double x = 3.14;
    printf("x = %d\n", x);   /* %d expects int, not double */
    return 0;
}
\end{lstlisting}
\begin{enumerate}[nosep,leftmargin=*]
  \item Does the compiler give a warning, an error, or nothing?
  \item What does the output look like? Is it what you expected?
  \item Change \texttt{\%d} to \texttt{\%f}. Does the output change?
  \item What rule does this teach about matching format specifiers to variable types?
\end{enumerate}
\end{tryit}

\begin{tryit}{Explore escape sequences}
In your ``Hello, world!'' program replace \texttt{\textbackslash n} with each of the
following one at a time, predict the output first, then run:
\begin{itemize}[nosep,leftmargin=*]
  \item \texttt{\textbackslash t} \quad (horizontal tab)
  \item \texttt{\textbackslash n\textbackslash n} \quad (two newlines)
  \item \texttt{\textbackslash\textbackslash} \quad (literal backslash)
  \item \texttt{\%\%} \quad (literal percent sign, inside the format string)
\end{itemize}
Record the one result that surprised you most.
\end{tryit}

\begin{tryit}{Integer division -- change one character, different result}
Start with:
\begin{lstlisting}
#include <stdio.h>
int main(void) {
    printf("%d\n", 7 / 2);
    return 0;
}
\end{lstlisting}
\begin{enumerate}[nosep,leftmargin=*]
  \item Run it. What is printed?
  \item Change \texttt{2} to \texttt{2.0}. What is printed now? Why?
  \item What format specifier do you need to use to see the decimal places?
\end{enumerate}
\end{tryit}

\section*{In-Lab Exercises (target: 2 hours)}

\begin{labexercise}{Hello + Your Info \hfill {\small \itshape (15 min)}}
Write a program that prints three lines:
(1) \texttt{Hello, world!}, (2) your full name, (3) your student ID.
\textbf{Goal:} verify the toolchain (compile + run) works end-to-end.
\end{labexercise}

\begin{labexercise}{Simple Calculator \hfill {\small \itshape (35 min)}}
Read two integers from the user and print their sum, difference, product, integer quotient,
and remainder.
\begin{itemize}[leftmargin=*,nosep]
  \item Label each line of output clearly, e.g.\ \texttt{Sum: 12 + 5 = 17}.
  \item What happens if the user enters \texttt{0} for the second number? Try it and note the behavior in a comment.
\end{itemize}
\end{labexercise}

\begin{labexercise}{Unit Conversion: Celsius $\to$ Fahrenheit \hfill {\small \itshape (30 min)}}
Read a temperature in Celsius (\texttt{double}) and print the equivalent in Fahrenheit using
$F = \tfrac{9}{5}C + 32$. Print the result with two digits after the decimal point
(\texttt{\%.2f}). Test with $C = 0$, $100$, and $-40$.
\end{labexercise}

\begin{labexercise}{ASCII Name Tag \hfill {\small \itshape (40 min)}}
Ask for the user's name (one word) and a single border character. Print a 3-line
ASCII box that contains the name. Example:
\begin{lstlisting}
##########
#  BMO   #
##########
\end{lstlisting}
\textbf{Hint:} use \texttt{scanf("\%s", name)} for the name and \texttt{scanf(" \%c", \&ch)} for the character.
\end{labexercise}

\begin{aicollab}
\textbf{Try these prompts with an AI tool:}
\begin{itemize}[leftmargin=*,nosep]
  \item ``Explain what each line of a minimal C program does:
        \texttt{\#include}, \texttt{main}, \texttt{return 0;}, and the semicolons.''
  \item ``What is the difference between \texttt{\%d}, \texttt{\%f}, and \texttt{\%lf} in
        \texttt{printf}? Give one example each.''
  \item ``My program compiles but prints garbage when I use \texttt{\%d} to print a
        \texttt{double}. Why?''
\end{itemize}

\medskip\textbf{Critical check -- verify, don't just accept:}
Take the AI's explanation of format specifiers and write a small program that
intentionally mismatches \texttt{\%d} with a \texttt{double}, \texttt{\%f} with an
\texttt{int}, and \texttt{\%s} with a \texttt{char}. Run it and compare the actual output
and compiler warnings against what the AI predicted. Note every discrepancy.

\medskip\textbf{Know the limit:}
AI tools explain format specifiers confidently but sometimes give example code that
compiles with a \emph{warning} rather than an error -- always read your compiler's full
output yourself, never assume the AI's snippet is warning-free.
\end{aicollab}

\begin{lstlisting}
#include <stdio.h>
int main(void) {
    int a = 7, b = 2;
