\section*{Objectives}
\begin{itemize}[leftmargin=*,nosep]
  \item Use \texttt{if}, \texttt{if-else}, and \texttt{if-else if-else} ladders to express multi-way decisions.
  \item Combine relational and logical operators (\texttt{\&\&}, \texttt{||}, \texttt{!}) into conditions.
  \item Choose between an \texttt{if}-ladder and a \texttt{switch} statement appropriately.
  \item Use \texttt{switch} fall-through deliberately, and remember the \texttt{break}.
\end{itemize}

\begin{summary}[Quick Reference]
\textbf{if-else.}
\begin{lstlisting}
if (cond) {
    /* runs when cond is non-zero */
} else if (other) {
    /* ... */
} else {
    /* fallback */
}
\end{lstlisting}

\textbf{switch.}
\begin{lstlisting}
switch (expr) {
    case 1: /* ... */ break;
    case 2:
    case 3: /* shared body */ break;
    default: /* ... */
}
\end{lstlisting}

\textbf{Logical operators} short-circuit:
\texttt{a \&\& b} skips \texttt{b} if \texttt{a} is false; \texttt{a || b} skips \texttt{b} if \texttt{a} is true.\\
\textbf{Common pitfall.} \texttt{=} is assignment; \texttt{==} is comparison.\\
\textbf{Range tests.} You cannot write \texttt{0 < x < 10}. Write \texttt{x > 0 \&\& x < 10}.
\end{summary}

\subsection*{Quick Experiments (Try It)}

\begin{tryit}{Assignment inside \texttt{if} -- the classic typo}
Run this program and observe:
\begin{lstlisting}
#include <stdio.h>
int main(void) {
    int x = 0;
    if (x = 5) {        /* assignment, not comparison! */
        printf("true branch\n");
    } else {
        printf("false branch\n");
    }
    printf("x is now: %d\n", x);
    return 0;
}
\end{lstlisting}
\begin{enumerate}[nosep,leftmargin=*]
  \item Which branch executes? What value does \texttt{x} have after the \texttt{if}?
  \item Change \texttt{x = 5} to \texttt{x = 0}. Which branch runs now? Why?
  \item Change to \texttt{x == 5}. Does it behave as you expect?
  \item Lesson: assignment \texttt{=} always evaluates to the assigned value;
        \texttt{if (a = 0)} is always false; \texttt{if (a = 1)} is always true.
\end{enumerate}
\end{tryit}

\begin{tryit}{Fall-through in \texttt{switch} -- remove one \texttt{break}}
Take your switch-based calculator from Exercise 3. Remove the \texttt{break} after
the \texttt{'+'} case only, then test the input \texttt{4 + 2}.
\begin{enumerate}[nosep,leftmargin=*]
  \item What is printed? Which cases ran?
  \item Now add \texttt{break} back and remove it after \texttt{'-'}. Test \texttt{4 - 2}.
  \item What is intentional fall-through useful for? (Hint: look at the take-home trace.)
\end{enumerate}
\end{tryit}

\begin{tryit}{Range check: the C chain-comparison trap}
Predict the output, then run:
\begin{lstlisting}
#include <stdio.h>
int main(void) {
    int x = 50;
    if (0 < x < 10)           /* legal C but NOT what you think */
        printf("in range\n");
    else
        printf("out of range\n");
    return 0;
}
\end{lstlisting}
\begin{enumerate}[nosep,leftmargin=*]
  \item Did the correct branch run? Explain what \texttt{0 < x < 10} actually means
        to the compiler. (Hint: evaluate \texttt{0 < x} first; what type is its result?)
  \item Fix the condition with \texttt{\&\&} and verify it now works correctly.
\end{enumerate}
\end{tryit}

\section*{In-Lab Exercises (target: 2 hours)}

\begin{labexercise}{Character Classification \hfill {\small \itshape (25 min)}}
Read a single character and report which category it belongs to:
\textit{digit}, \textit{lowercase letter}, \textit{uppercase letter}, \textit{space}, or
\textit{other}. Use an \texttt{if-else if} ladder, not \texttt{ctype.h}.
\end{labexercise}

\begin{labexercise}{Leap Year \hfill {\small \itshape (25 min)}}
Read a year and print whether it is a leap year. A year is a leap year if it is
divisible by 4 \emph{and} (not divisible by 100 \emph{or} divisible by 400). Express
this in a single boolean expression using \texttt{\&\&} and \texttt{||}.
\end{labexercise}

\begin{labexercise}{Simple Calculator with \texttt{switch} \hfill {\small \itshape (35 min)}}
Read two doubles and a single-character operator (\texttt{+}, \texttt{-}, \texttt{*}, \texttt{/})
and print the result. Use a \texttt{switch} on the operator. Handle division by zero
gracefully (print an error message instead of dividing).
\end{labexercise}

\begin{labexercise}{Triangle Type \hfill {\small \itshape (35 min)}}
Read three positive numbers as side lengths. First, check the triangle inequality:
each side must be shorter than the sum of the other two. If invalid, print
\texttt{Not a triangle}. Otherwise, classify as \textit{Equilateral}, \textit{Isosceles},
or \textit{Scalene}.
\end{labexercise}

\begin{aicollab}
\textbf{Try these prompts with an AI tool:}
\begin{itemize}[leftmargin=*,nosep]
  \item ``Explain the dangling-else problem in C with a concrete example and show how
        braces fix it.''
  \item ``When should I use \texttt{switch} instead of an \texttt{if-else if} ladder?
        What are the restrictions on \texttt{switch} expressions in C?''
  \item ``Why does \texttt{if (x = 5)} always enter the true branch? What compiler
        warning catches this?''
\end{itemize}

\medskip\textbf{Critical check -- verify, don't just accept:}
Ask the AI to write a grade classifier using \texttt{switch} (A/B/C/D/F from a score).
Before running, count how many \texttt{break} statements appear and whether every case
has one. Compile and test with edge inputs: exactly 90, exactly 100, 0, and $-1$.
Record any behavior that differs from the AI's stated output.

\medskip\textbf{Know the limit:}
AI tools regularly omit or misplace \texttt{break} statements in \texttt{switch} blocks
and rarely distinguish intentional fall-through from accidental fall-through -- inspect
every case boundary yourself.
\end{aicollab}

\begin{trace}[Dangling else]

\begin{lstlisting}
#include <stdio.h>
int main(void) {
    int x = 0;
    if (x > 0)
        if (x > 10) printf("big\n");
    else printf("small\n");
    return 0;
}
\end{lstlisting}
\textbf{Hint:} an \texttt{else} associates with the nearest unmatched \texttt{if}.
\end{trace}

\begin{trace}[\texttt{switch} fall-through]
Predict the output for inputs \texttt{1}, \texttt{2}, and \texttt{3}:
\begin{lstlisting}
#include <stdio.h>
int main(void) {
    int n; scanf("%d", &n);
    switch (n) {
        case 1: printf("one ");
        case 2: printf("two "); break;
        case 3: printf("three ");
        default: printf("end");
    }
    printf("\n");
    return 0;
}
\end{lstlisting}
\end{trace}
